480 # A type variable is uniquely identified by its raw id and meta level.
481  
482 # For plain variables (type parameters of generic classes and
483 # functions) raw ids are allocated by semantic analysis, using
484 # positive ids 1, 2, ... for generic class parameters and negative
485 # ids -1, ... for generic function type arguments. A special value 0
486 # is reserved for Self type variable (autogenerated). This convention
487 # is only used to keep type variable ids distinct when allocating
488 # them; the type checker makes no distinction between class and
489 # function type variables.
490  
491 # Metavariables are allocated unique ids starting from 1.
492 raw_id: int = 0
493  
494 # Level of the variable in type inference. Currently either 0 for
495 # declared types, or 1 for type inference metavariables.

499 next_raw_id: ClassVar[int] = 1

502 # declares this type variable (not the fullname of the TypeVar
503 # definition!), or ''

1396 # have last_known_values of type Literal[1], Literal[2], Literal[3], and Literal[4]
1397 # respectively. However, the Instance object assigned to 'a' and 'b' will have their
1398 # last_known_value erased: variable 'a' is mutable; variable 'b' was declared to be
1399 # specifically an int.
1400 #
1401 # Or more broadly, this field lets this Instance "remember" its original declaration
1402 # when applicable. We want this behavior because we want implicit Final declarations
1403 # to act pretty much identically with constants: we should be able to replace any
1404 # places where we use some Final variable with the original value and get the same
1405 # type-checking behavior. For example, we want this program:
1406 #
1407 # def expects_literal(x: Literal[3]) -> None: pass
1408 # var: Final = 3
1409 # expects_literal(var)
1410 #
1411 # ...to type-check in the exact same way as if we had written the program like this:
1412 #
1413 # def expects_literal(x: Literal[3]) -> None: pass
1414 # expects_literal(3)
1415 #
1416 # In order to make this work (especially with literal types), we need var's type
1417 # (an Instance) to remember the "original" value.
1418 #
1419 # Preserving this value within expressions is useful for similar reasons.
1420 #
1421 # Currently most of mypy will ignore this field and will continue to treat this type like
1422 # a regular Instance. We end up using this field only when we are explicitly within a
