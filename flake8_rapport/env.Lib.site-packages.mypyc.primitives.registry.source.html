1 """Utilities for defining primitive ops.
2  
3 Most of the ops can be automatically generated by matching against AST
4 nodes and types. For example, a func_op is automatically generated when
5 a specific function is called with the specific positional argument
6 count and argument types.
7  
8 Example op definition:
9  
10 list_len_op = func_op(name='builtins.len',
11 arg_types=[list_rprimitive],
12 result_type=short_int_rprimitive,
13 error_kind=ERR_NEVER,
14 emit=emit_len)
15  
16 This op is automatically generated for calls to len() with a single
17 list argument. The result type is short_int_rprimitive, and this
18 never raises an exception (ERR_NEVER). The function emit_len is used
19 to generate C for this op. The op can also be manually generated using
20 "list_len_op". Ops that are only generated automatically don't need to
21 be assigned to a module attribute.
22  
23 Ops defined with custom_op are only explicitly generated in
24 mypyc.irbuild and won't be generated automatically. They are always
25 assigned to a module attribute, as otherwise they won't be accessible.
26  
27 The actual ops are defined in other submodules of this package, grouped
28 by category.
29  
30 Most operations have fallback implementations that apply to all possible
31 arguments and types. For example, there are generic implementations of
32 arbitrary function and method calls, and binary operators. These generic
33 implementations are typically slower than specialized ones, but we tend
34 to rely on them for infrequently used ops. It's impractical to have
35 optimized implementations of all ops.
36 """

87 def method_op(
88 name: str,
89 arg_types: list[RType],
90 return_type: RType,
91 c_function_name: str,
92 error_kind: int,
93 var_arg_type: RType | None = None,
94 truncated_type: RType | None = None,
95 ordering: list[int] | None = None,
96 extra_int_constants: list[tuple[int, RType]] = [],
97 steals: StealsDescription = False,
98 is_borrowed: bool = False,
99 priority: int = 1,
100 ) -> CFunctionDescription:
101 """Define a c function call op that replaces a method call.
102  
103 This will be automatically generated by matching against the AST.
104  
105 Args:
106 name: short name of the method (for example, 'append')
107 arg_types: argument types; the receiver is always the first argument
108 return_type: type of the return value. Use void_rtype to represent void.
109 c_function_name: name of the C function to call
110 error_kind: how errors are represented in the result (one of ERR_*)
111 var_arg_type: type of all variable arguments
112 truncated_type: type to truncated to(See Truncate for info)
113 if it's defined both return_type and it should be non-referenced
114 integer types or bool type
115 ordering: optional ordering of the arguments, if defined,
116 reorders the arguments accordingly.
117 should never be used together with var_arg_type.
118 all the other arguments(such as arg_types) are in the order
119 accepted by the python syntax(before reordering)
120 extra_int_constants: optional extra integer constants as the last arguments to a C call
121 steals: description of arguments that this steals (ref count wise)
122 is_borrowed: if True, returned value is borrowed (no need to decrease refcount)
123 priority: if multiple ops match, the one with the highest priority is picked
124 """
125 ops = method_call_ops.setdefault(name, [])
126 desc = CFunctionDescription(
127 name,
128 arg_types,
129 return_type,
130 var_arg_type,
131 truncated_type,
132 c_function_name,
133 error_kind,
134 steals,
135 is_borrowed,
136 ordering,
137 extra_int_constants,
138 priority,
139 )
140 ops.append(desc)
141 return desc

298 def load_address_op(name: str, type: RType, src: str) -> LoadAddressDescription:
299 assert name not in builtin_names, "already defined: %s" % name
300 builtin_names[name] = (type, src)
