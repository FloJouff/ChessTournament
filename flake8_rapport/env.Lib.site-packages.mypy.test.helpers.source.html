217 def clean_up(a: list[str]) -> list[str]:
218 """Remove common directory prefix from all strings in a.
219  
220 This uses a naive string replace; it seems to work well enough. Also
221 remove trailing carriage returns.
222 """
223 res = []
224 pwd = os.getcwd()
225 driver = pwd + "/driver.py"
226 for s in a:
227 prefix = os.sep
228 ss = s
229 for p in prefix, prefix.replace(os.sep, "/"):
230 if p != "/" and p != "//" and p != "\\" and p != "\\\\":
231 ss = ss.replace(p, "")
232 # Ignore spaces at end of line.
233 ss = re.sub(" +$", "", ss)
234 # Remove pwd from driver.py's path
235 ss = ss.replace(driver, "driver.py")
236 res.append(re.sub("\\r$", "", ss))
237 return res

280 def retry_on_error(func: Callable[[], Any], max_wait: float = 1.0) -> None:
281 """Retry callback with exponential backoff when it raises OSError.
282  
283 If the function still generates an error after max_wait seconds, propagate
284 the exception.
285  
286 This can be effective against random file system operation failures on
287 Windows.
288 """
289 t0 = time.time()
290 wait_time = 0.01
291 while True:
292 try:
293 func()
294 return
295 except OSError:
296 wait_time = min(wait_time * 2, t0 + max_wait - time.time())
297 if wait_time <= 0.01:
298 # Done enough waiting, the error seems persistent.
299 raise
300 time.sleep(wait_time)

377 def write_and_fudge_mtime(content: str, target_path: str) -> None:
378 # In some systems, mtime has a resolution of 1 second which can
379 # cause annoying-to-debug issues when a file has the same size
380 # after a change. We manually set the mtime to circumvent this.
381 # Note that we increment the old file's mtime, which guarantees a
382 # different value, rather than incrementing the mtime after the
383 # copy, which could leave the mtime unchanged if the old file had
384 # a similarly fudged mtime.
385 new_time = None
386 if os.path.isfile(target_path):
387 new_time = os.stat(target_path).st_mtime + 1
388  
389 dir = os.path.dirname(target_path)
390 os.makedirs(dir, exist_ok=True)
391 with open(target_path, "w", encoding="utf-8") as target:
392 target.write(content)
393  
394 if new_time:
395 os.utime(target_path, times=(new_time, new_time))

399 for op in operations:
400 if isinstance(op, UpdateFile):
401 # Modify/create file
402 write_and_fudge_mtime(op.content, op.target_path)

404 # Delete file/directory
405 if os.path.isdir(op.path):
406 # Sanity check to avoid unexpected deletions

410 # Use retries to work around potential flakiness on Windows (AppVeyor).
411 path = op.path

457 def normalize_file_output(content: list[str], current_abs_path: str) -> list[str]:
458 """Normalize file output for comparison."""
459 timestamp_regex = re.compile(r"\d{10}")
460 result = [x.replace(current_abs_path, "$PWD") for x in content]
461 version = mypy.version.__version__
462 result = [re.sub(r"\b" + re.escape(version) + r"\b", "$VERSION", x) for x in result]
463 # We generate a new mypy.version when building mypy wheels that
464 # lacks base_version, so handle that case.
465 base_version = getattr(mypy.version, "base_version", version)
466 result = [re.sub(r"\b" + re.escape(base_version) + r"\b", "$VERSION", x) for x in result]
467 result = [timestamp_regex.sub("$TIMESTAMP", x) for x in result]
468 return result
