123 def parse_args(line: str) -> list[str]:
124 """Parse the first line of the program for the command line.
125  
126 This should have the form
127  
128 # cmd: mypy <options>
129  
130 For example:
131  
132 # cmd: mypy pkg/
133 """
134 m = re.match("# cmd: mypy (.*)$", line)
135 if not m:
136 return [] # No args; mypy will spit out an error.
137 return m.group(1).split()
