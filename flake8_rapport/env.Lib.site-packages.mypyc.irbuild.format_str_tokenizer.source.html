32 class FormatOp(Enum):
33 """FormatOp represents conversion operations of string formatting during
34 compile time.
35  
36 Compare to ConversionSpecifier, FormatOp has fewer attributes.
37 For example, to mark a conversion from any object to string,
38 ConversionSpecifier may have several representations, like '%s', '{}'
39 or '{:{}}'. However, there would only exist one corresponding FormatOp.
40 """
41  
42 STR = "s"
43 INT = "d"
44 BYTES = "b"

47 def generate_format_ops(specifiers: list[ConversionSpecifier]) -> list[FormatOp] | None:
48 """Convert ConversionSpecifier to FormatOp.
49  
50 Different ConversionSpecifiers may share a same FormatOp.
51 """
52 format_ops = []
53 for spec in specifiers:
54 # TODO: Match specifiers instead of using whole_seq
55 if spec.whole_seq == "%s" or spec.whole_seq == "{:{}}":
56 format_op = FormatOp.STR
57 elif spec.whole_seq == "%d":
58 format_op = FormatOp.INT
59 elif spec.whole_seq == "%b":
60 format_op = FormatOp.BYTES
61 elif spec.whole_seq:
62 return None
63 else:
64 format_op = FormatOp.STR
65 format_ops.append(format_op)
66 return format_ops

69 def tokenizer_printf_style(format_str: str) -> tuple[list[str], list[FormatOp]] | None:
70 """Tokenize a printf-style format string using regex.
71  
72 Return:
73 A list of string literals and a list of FormatOps.
74 """
75 literals: list[str] = []
76 specifiers: list[ConversionSpecifier] = parse_conversion_specifiers(format_str)
77 format_ops = generate_format_ops(specifiers)
78 if format_ops is None:
79 return None
80  
81 last_end = 0
82 for spec in specifiers:
83 cur_start = spec.start_pos
84 literals.append(format_str[last_end:cur_start])
85 last_end = cur_start + len(spec.whole_seq)
86 literals.append(format_str[last_end:])
87  
88 return literals, format_ops

96 def tokenizer_format_call(format_str: str) -> tuple[list[str], list[FormatOp]] | None:
97 """Tokenize a str.format() format string.
98  
99 The core function parse_format_value() is shared with mypy.
100 With these specifiers, we then parse the literal substrings
101 of the original format string and convert `ConversionSpecifier`
102 to `FormatOp`.
103  
104 Return:
105 A list of string literals and a list of FormatOps. The literals
106 are interleaved with FormatOps and the length of returned literals
107 should be exactly one more than FormatOps.
108 Return None if it cannot parse the string.
109 """
110 # Creates an empty MessageBuilder here.
111 # It wouldn't be used since the code has passed the type-checking.
112 specifiers = parse_format_value(
113 format_str, EMPTY_CONTEXT, MessageBuilder(Errors(Options()), {})
114 )
115 if specifiers is None:
116 return None
117 format_ops = generate_format_ops(specifiers)
118 if format_ops is None:
119 return None
120  
121 literals: list[str] = []
122 last_end = 0
123 for spec in specifiers:
124 # Skip { and }
  • E203 Whitespace before ':'
125 literals.append(format_str[last_end : spec.start_pos - 1])
126 last_end = spec.start_pos + len(spec.whole_seq) + 1
127 literals.append(format_str[last_end:])
128 # Deal with escaped {{
129 literals = [x.replace("{{", "{").replace("}}", "}") for x in literals]
130  
131 return literals, format_ops

134 def convert_format_expr_to_str(
135 builder: IRBuilder, format_ops: list[FormatOp], exprs: list[Expression], line: int
136 ) -> list[Value] | None:
137 """Convert expressions into string literal objects with the guidance
138 of FormatOps. Return None when fails."""
139 if len(format_ops) != len(exprs):
140 return None
141  
142 converted = []
143 for x, format_op in zip(exprs, format_ops):
144 node_type = builder.node_type(x)
145 if format_op == FormatOp.STR:
146 if is_str_rprimitive(node_type):
147 var_str = builder.accept(x)
148 elif is_int_rprimitive(node_type) or is_short_int_rprimitive(node_type):
149 var_str = builder.call_c(int_to_str_op, [builder.accept(x)], line)
150 else:
151 var_str = builder.call_c(str_op, [builder.accept(x)], line)
152 elif format_op == FormatOp.INT:
153 if is_int_rprimitive(node_type) or is_short_int_rprimitive(node_type):
154 var_str = builder.call_c(int_to_str_op, [builder.accept(x)], line)
155 else:
156 return None
157 else:
158 return None
159 converted.append(var_str)
160 return converted

163 def join_formatted_strings(
164 builder: IRBuilder, literals: list[str] | None, substitutions: list[Value], line: int
165 ) -> Value:
166 """Merge the list of literals and the list of substitutions
167 alternatively using 'str_build_op'.
168  
169 `substitutions` is the result value of formatting conversions.
170  
171 If the `literals` is set to None, we simply join the substitutions;
172 Otherwise, the `literals` is the literal substrings of the original
173 format string and its length should be exactly one more than
174 substitutions.
175  
176 For example:
177 (1) 'This is a %s and the value is %d'
178 -> literals: ['This is a ', ' and the value is', '']
179 (2) '{} and the value is {}'
180 -> literals: ['', ' and the value is', '']
181 """
182 # The first parameter for str_build_op is the total size of
183 # the following PyObject*
184 result_list: list[Value] = [Integer(0, c_pyssize_t_rprimitive)]
185  
186 if literals is not None:
187 for a, b in zip(literals, substitutions):
188 if a:
189 result_list.append(builder.load_str(a))
190 result_list.append(b)
191 if literals[-1]:
192 result_list.append(builder.load_str(literals[-1]))
193 else:
194 result_list.extend(substitutions)
195  
196 # Special case for empty string and literal string
197 if len(result_list) == 1:
198 return builder.load_str("")
199 if not substitutions and len(result_list) == 2:
200 return result_list[1]
201  
202 result_list[0] = Integer(len(result_list) - 1, c_pyssize_t_rprimitive)
203 return builder.call_c(str_build_op, result_list, line)

229 def join_formatted_bytes(
230 builder: IRBuilder, literals: list[str], substitutions: list[Value], line: int
231 ) -> Value:
232 """Merge the list of literals and the list of substitutions
233 alternatively using 'bytes_build_op'."""
234 result_list: list[Value] = [Integer(0, c_pyssize_t_rprimitive)]
235  
236 for a, b in zip(literals, substitutions):
237 if a:
238 result_list.append(builder.load_bytes_from_str_literal(a))
