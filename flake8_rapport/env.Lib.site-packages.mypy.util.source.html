118 def find_python_encoding(text: bytes) -> tuple[str, int]:
119 """PEP-263 for detecting Python file encoding"""
120 result = ENCODING_RE.match(text)
121 if result:
122 line = 2 if result.group(1) else 1
123 encoding = result.group(3).decode("ascii")
124 # Handle some aliases that Python is happy to accept and that are used in the wild.
125 if encoding.startswith(("iso-latin-1-", "latin-1-")) or encoding == "iso-latin-1":
126 encoding = "latin-1"
127 return encoding, line
128 else:
129 default_encoding = "utf8"
130 return default_encoding, -1

154 def decode_python_encoding(source: bytes) -> str:
155 """Read the Python file with while obeying PEP-263 encoding detection.
156  
157 Returns the source as a string.
158 """
159 # check for BOM UTF-8 encoding and strip it out if present
160 if source.startswith(b"\xef\xbb\xbf"):
161 encoding = "utf8"
162 source = source[3:]
163 else:
164 # look at first two lines and check if PEP-263 coding is present
165 encoding, _ = find_python_encoding(source)

168 source_text = source.decode(encoding)
169 except LookupError as lookuperr:
170 raise DecodeError(str(lookuperr)) from lookuperr

191 def trim_source_line(line: str, max_len: int, col: int, min_width: int) -> tuple[str, int]:
192 """Trim a line of source code to fit into max_len.
193  
194 Show 'min_width' characters on each side of 'col' (an error location). If either
195 start or end is trimmed, this is indicated by adding '...' there.
196 A typical result looks like this:
197 ...some_variable = function_to_call(one_arg, other_arg) or...
198  
199 Return the trimmed string and the column offset to to adjust error location.
200 """
201 if max_len < 2 * min_width + 1:
202 # In case the window is too tiny it is better to still show something.
203 max_len = 2 * min_width + 1
204  
205 # Trivial case: line already fits in.
206 if len(line) <= max_len:
207 return line, 0
208  
209 # If column is not too large so that there is still min_width after it,
210 # the line doesn't need to be trimmed at the start.
211 if col + min_width < max_len:
212 return line[:max_len] + "...", 0
213  
214 # Otherwise, if the column is not too close to the end, trim both sides.
215 if col < len(line) - min_width - 1:
216 offset = col - max_len + min_width + 1
  • E203 Whitespace before ':'
217 return "..." + line[offset : col + min_width + 1] + "...", offset - 3
218  
219 # Finally, if the column is near the end, just trim the start.
220 return "..." + line[-max_len:], len(line) - max_len - 3

370 import inspect # Lazy import for minor startup speed win

373 # (that is, attributes from __slots__ and C extension classes)

494 def soft_wrap(msg: str, max_len: int, first_offset: int, num_indent: int = 0) -> str:
495 """Wrap a long error message into few lines.
496  
497 Breaks will only happen between words, and never inside a quoted group
498 (to avoid breaking types such as "Union[int, str]"). The 'first_offset' is
499 the width before the start of first line.
500  
501 Pad every next line with 'num_indent' spaces. Every line will be at most 'max_len'
502 characters, except if it is a single word or quoted group.
503  
504 For example:
505 first_offset
506 ------------------------
507 path/to/file: error: 58: Some very long error message
508 that needs to be split in separate lines.
509 "Long[Type, Names]" are never split.
510 ^^^^--------------------------------------------------
511 num_indent max_len
512 """
513 words = split_words(msg)
514 next_line = words.pop(0)
515 lines: list[str] = []
516 while words:
517 next_word = words.pop(0)
518 max_line_len = max_len - num_indent if lines else max_len - first_offset
519 # Add 1 to account for space between words.
520 if len(next_line) + len(next_word) + 1 <= max_line_len:
521 next_line += " " + next_word
522 else:
523 lines.append(next_line)
524 next_line = next_word
525 lines.append(next_line)
526 padding = "\n" + " " * num_indent
527 return padding.join(lines)

695 width = fixed_terminal_width or get_terminal_width()

697 for i, error in enumerate(messages):
698 if ": error:" in error:
699 loc, msg = error.split("error:", maxsplit=1)
700 msg = soft_wrap(msg, width, first_offset=len(loc) + len("error: "))
701 new_messages[i] = loc + "error:" + msg
702 if error.startswith(" " * DEFAULT_SOURCE_OFFSET) and "^" not in error:
703 # TODO: detecting source code highlights through an indent can be surprising.
704 # Restore original error message and error location.
705 error = error[DEFAULT_SOURCE_OFFSET:]
706 marker_line = messages[i + 1]
707 marker_column = marker_line.index("^")
708 column = marker_column - DEFAULT_SOURCE_OFFSET
709 if "~" not in marker_line:
710 marker = "^"
711 else:
712 # +1 because both ends are included
  • E203 Whitespace before ':'
713 marker = marker_line[marker_column : marker_line.rindex("~") + 1]
714  
715 # Let source have some space also on the right side, plus 6
716 # to accommodate ... on each side.
717 max_len = width - DEFAULT_SOURCE_OFFSET - 6
718 source_line, offset = trim_source_line(error, max_len, column, MINIMUM_WIDTH)
719  
720 new_messages[i] = " " * DEFAULT_SOURCE_OFFSET + source_line
721 # Also adjust the error marker position and trim error marker is needed.
722 new_marker_line = " " * (DEFAULT_SOURCE_OFFSET + column - offset) + marker
723 if len(new_marker_line) > len(new_messages[i]) and len(marker) > 3:
724 new_marker_line = new_marker_line[: len(new_messages[i]) - 3] + "..."
725 new_messages[i + 1] = new_marker_line
726 return new_messages

728 def colorize(self, error: str) -> str:
729 """Colorize an output line by highlighting the status and error code."""
730 if ": error:" in error:
731 loc, msg = error.split("error:", maxsplit=1)
732 if self.hide_error_codes:
733 return (
734 loc + self.style("error:", "red", bold=True) + self.highlight_quote_groups(msg)
735 )
736 codepos = msg.rfind("[")
737 if codepos != -1:
738 code = msg[codepos:]
739 msg = msg[:codepos]
740 else:
741 code = "" # no error code specified
742 return (
743 loc
744 + self.style("error:", "red", bold=True)
745 + self.highlight_quote_groups(msg)
746 + self.style(code, "yellow")
747 )
748 elif ": note:" in error:
749 loc, msg = error.split("note:", maxsplit=1)
750 formatted = self.highlight_quote_groups(self.underline_link(msg))
751 return loc + self.style("note:", "blue") + formatted
752 elif error.startswith(" " * DEFAULT_SOURCE_OFFSET):
753 # TODO: detecting source code highlights through an indent can be surprising.
754 if "^" not in error:
755 return self.style(error, "none", dim=True)
756 return self.style(error, "red")
757 else:
758 return error

760 def highlight_quote_groups(self, msg: str) -> str:
761 """Make groups quoted with double quotes bold (including quotes).
762  
763 This is used to highlight types, attribute names etc.
764 """
765 if msg.count('"') % 2:
766 # Broken error message, don't do any formatting.
767 return msg
768 parts = msg.split('"')
769 out = ""
770 for i, part in enumerate(parts):
771 if i % 2 == 0:
772 out += self.style(part, "none")
773 else:
774 out += self.style('"' + part + '"', "none", bold=True)
775 return out

777 def underline_link(self, note: str) -> str:
778 """Underline a link in a note message (if any).
779  
780 This assumes there is at most one link in the message.
781 """
782 match = re.search(r"https?://\S*", note)
783 if not match:
784 return note
785 start = match.start()
786 end = match.end()
787 return note[:start] + self.style(note[start:end], "none", underline=True) + note[end:]

854 def quote_docstring(docstr: str) -> str:
855 """Returns docstring correctly encapsulated in a single or double quoted form."""
856 # Uses repr to get hint on the correct quotes and escape everything properly.
857 # Creating multiline string for prettier output.
858 docstr_repr = "\n".join(re.split(r"(?<=[^\\])\\n", repr(docstr)))
859  
860 if docstr_repr.startswith("'"):
861 # Enforce double quotes when it's safe to do so.
862 # That is when double quotes are not in the string
863 # or when it doesn't end with a single quote.
864 if '"' not in docstr_repr[1:-1] and docstr_repr[-2] != "'":
865 return f'"""{docstr_repr[1:-1]}"""'
866 return f"''{docstr_repr}''"
867 else:
868 return f'""{docstr_repr}""'
