100 ) -> ModuleIR:

103 # By default generate IR compatible with the earliest supported Python C API.
104 # If a test needs more recent API features, this should be overridden.
105 compiler_options = compiler_options or CompilerOptions(capi_version=(3, 7))
106 options = Options()
107 options.show_traceback = True
108 options.hide_error_codes = True
109 options.use_builtins_fixtures = True
110 options.strict_optional = True
111 options.python_version = compiler_options.python_version or (3, 6)
112 options.export_types = True
113 options.preserve_asts = True
114 options.allow_empty_bodies = True
115 options.per_module_options["__main__"] = {"mypyc": True}

117 source = build.BuildSource("main", "__main__", program_text)
118 # Construct input as a single single.

122 raise CompileError(result.errors)

238 def replace_word_size(text: list[str]) -> list[str]:
239 """Replace WORDSIZE with platform specific word sizes"""
240 result = []
241 for line in text:
242 index = line.find("WORD_SIZE")
243 if index != -1:
244 # get 'WORDSIZE*n' token
245 word_size_token = line[index:].split()[0]
246 n = int(word_size_token[10:])
247 replace_str = str(PLATFORM_SIZE * n)
248 result.append(line.replace(word_size_token, replace_str))
249 else:
250 result.append(line)
251 return result

254 def infer_ir_build_options_from_test_name(name: str) -> CompilerOptions | None:
255 """Look for magic substrings in test case name to set compiler options.
256  
257 Return None if the test case should be skipped (always pass).
258  
259 Supported naming conventions:
260  
261 *_64bit*:
262 Run test case only on 64-bit platforms
263 *_32bit*:
264 Run test caseonly on 32-bit platforms
265 *_python3_8* (or for any Python version):
266 Use Python 3.8+ C API features (default: lowest supported version)
267 *StripAssert*:
268 Don't generate code for assert statements
269 """
270 # If this is specific to some bit width, always pass if platform doesn't match.
271 if "_64bit" in name and IS_32_BIT_PLATFORM:
272 return None
273 if "_32bit" in name and not IS_32_BIT_PLATFORM:
274 return None
275 options = CompilerOptions(strip_asserts="StripAssert" in name, capi_version=(3, 7))
276 # A suffix like _python3.8 is used to set the target C API version.
277 m = re.search(r"_python([3-9]+)_([0-9]+)(_|\b)", name)
278 if m:
279 options.capi_version = (int(m.group(1)), int(m.group(2)))
280 options.python_version = options.capi_version
281 elif "_py" in name or "_Python" in name:
282 assert False, f"Invalid _py* suffix (should be _pythonX_Y): {name}"
283 return options
