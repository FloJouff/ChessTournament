6 class NameGenerator:
7 """Utility for generating distinct C names from Python names.
8  
9 Since C names can't use '.' (or unicode), some care is required to
10 make C names generated from Python names unique. Also, we want to
11 avoid generating overly long C names since they make the generated
12 code harder to read.
13  
14 Note that we don't restrict ourselves to a 32-character distinguishing
15 prefix guaranteed by the C standard since all the compilers we care
16 about at the moment support longer names without issues.
17  
18 For names that are exported in a shared library (not static) use
19 exported_name() instead.
20  
21 Summary of the approach:
22  
23 * Generate a unique name prefix from suffix of fully-qualified
24 module name used for static names. If only compiling a single
25 module, this can be empty. For example, if the modules are
26 'foo.bar' and 'foo.baz', the prefixes can be 'bar_' and 'baz_',
27 respectively. If the modules are 'bar.foo' and 'baz.foo', the
28 prefixes will be 'bar_foo_' and 'baz_foo_'.
29  
30 * Replace '.' in the Python name with '___' in the C name. (And
31 replace the unlikely but possible '___' with '___3_'. This
32 collides '___' with '.3_', but this is OK because names
33 may not start with a digit.)
34  
35 The generated should be internal to a build and thus the mapping is
36 arbitrary. Just generating names '1', '2', ... would be correct,
37 though not very usable.
38 """

40 def __init__(self, groups: Iterable[list[str]]) -> None:
41 """Initialize with a list of modules in each compilation group.
42  
43 The names of modules are used to shorten names referring to

54 def private_name(self, module: str, partial_name: str | None = None) -> str:
55 """Return a C name usable for a static definition.
56  
57 Return a distinct result for each (module, partial_name) pair.
58  
59 The caller should add a suitable prefix to the name to avoid
60 conflicts with other C names. Only ensure that the results of
61 this function are unique, not that they aren't overlapping with
62 arbitrary names.
63  
64 If a name is not specific to any module, the module argument can
65 be an empty string.
66 """
67 # TODO: Support unicode
68 if partial_name is None:
69 return exported_name(self.module_map[module].rstrip("."))
70 if (module, partial_name) in self.translations:
71 return self.translations[module, partial_name]
72 if module in self.module_map:
73 module_prefix = self.module_map[module]
74 elif module:
75 module_prefix = module + "."
76 else:
77 module_prefix = ""
78 actual = exported_name(f"{module_prefix}{partial_name}")
79 self.translations[module, partial_name] = actual
80 return actual

94 def make_module_translation_map(names: list[str]) -> dict[str, str]:
95 num_instances: dict[str, int] = {}
96 for name in names:
97 for suffix in candidate_suffixes(name):
98 num_instances[suffix] = num_instances.get(suffix, 0) + 1
99 result = {}
100 for name in names:
101 for suffix in candidate_suffixes(name):
102 if num_instances[suffix] == 1:
103 result[name] = suffix
104 break
105 else:
106 assert False, names
107 return result
