32 def walk_packages(
33 inspect: ModuleInspect, packages: list[str], verbose: bool = False
34 ) -> Iterator[str]:
35 """Iterates through all packages and sub-packages in the given list.
36  
37 This uses runtime imports (in another process) to find both Python and C modules.
38 For Python packages we simply pass the __path__ attribute to pkgutil.walk_packages() to
39 get the content of the package (all subpackages and modules). However, packages in C
40 extensions do not have this attribute, so we have to roll out our own logic: recursively
41 find all modules imported in the package that have matching names.
42 """
43 for package_name in packages:
44 if package_name in NOT_IMPORTABLE_MODULES:
45 print(f"{package_name}: Skipped (blacklisted)")
46 continue
47 if verbose:
48 print(f"Trying to import {package_name!r} for runtime introspection")
49 try:
50 prop = inspect.get_package_properties(package_name)
51 except InspectError:
52 report_missing(package_name)
53 continue
54 yield prop.name
55 if prop.is_c_module:
56 # Recursively iterate through the subpackages
57 yield from walk_packages(inspect, prop.subpackages, verbose)
58 else:
59 yield from prop.subpackages

152 def remove_misplaced_type_comments(source: str | bytes) -> str | bytes:
153 """Remove comments from source that could be understood as misplaced type comments.
154  
155 Normal comments may look like misplaced type comments, and since they cause blocking
156 parse errors, we want to avoid them.
157 """
158 if isinstance(source, bytes):
159 # This gives us a 1-1 character code mapping, so it's roundtrippable.
160 text = source.decode("latin1")
161 else:
162 text = source
163  
164 # Remove something that looks like a variable type comment but that's by itself
165 # on a line, as it will often generate a parse error (unless it's # type: ignore).
166 text = re.sub(r'^[ \t]*# +type: +["\'a-zA-Z_].*$', "", text, flags=re.MULTILINE)
167  
168 # Remove something that looks like a function type comment after docstring,
169 # which will result in a parse error.
170 text = re.sub(r'""" *\n[ \t\n]*# +type: +\(.*$', '"""\n', text, flags=re.MULTILINE)
171 text = re.sub(r"''' *\n[ \t\n]*# +type: +\(.*$", "'''\n", text, flags=re.MULTILINE)
172  
173 # Remove something that looks like a badly formed function type comment.
174 text = re.sub(r"^[ \t]*# +type: +\([^()]+(\)[ \t]*)?$", "", text, flags=re.MULTILINE)
175  
176 if isinstance(source, bytes):
177 return text.encode("latin1")
178 else:
179 return text

395 def __init__(self) -> None:
396 # module_for['foo'] has the module name where 'foo' was imported from, or None if
397 # 'foo' is a module imported directly;
398 # direct_imports['foo'] is the module path used when the name 'foo' was added to the
399 # namespace.
400 # reverse_alias['foo'] is the name that 'foo' had originally when imported with an
401 # alias; examples
402 # 'from pkg import mod' ==> module_for['mod'] == 'pkg'
403 # 'from pkg import mod as m' ==> module_for['m'] == 'pkg'
404 # ==> reverse_alias['m'] == 'mod'
405 # 'import pkg.mod as m' ==> module_for['m'] == None
406 # ==> reverse_alias['m'] == 'pkg.mod'
407 # 'import pkg.mod' ==> module_for['pkg'] == None
408 # ==> module_for['pkg.mod'] == None
409 # ==> direct_imports['pkg'] == 'pkg.mod'
410 # ==> direct_imports['pkg.mod'] == 'pkg.mod'
411 self.module_for: dict[str, str | None] = {}
412 self.direct_imports: dict[str, str] = {}
413 self.reverse_alias: dict[str, str] = {}
414  
415 # required_names is the set of names that are actually used in a type annotation
416 self.required_names: set[str] = set()
417  
418 # Names that should be reexported if they come from another module
419 self.reexports: set[str] = set()

421 def add_import_from(
422 self, module: str, names: list[tuple[str, str | None]], require: bool = False
423 ) -> None:
424 for name, alias in names:
425 if alias:
426 # 'from {module} import {name} as {alias}'
427 self.module_for[alias] = module
428 self.reverse_alias[alias] = name
429 else:
430 # 'from {module} import {name}'
431 self.module_for[name] = module
432 self.reverse_alias.pop(name, None)
433 if require:
434 self.require_name(alias or name)
435 self.direct_imports.pop(alias or name, None)

437 def add_import(self, module: str, alias: str | None = None, require: bool = False) -> None:
438 if alias:
439 # 'import {module} as {alias}'
440 assert "." not in alias # invalid syntax
441 self.module_for[alias] = None
442 self.reverse_alias[alias] = module
443 if require:
444 self.required_names.add(alias)
445 else:
446 # 'import {module}'
447 name = module
448 if require:
449 self.required_names.add(name)
450 # add module and its parent packages
451 while name:
452 self.module_for[name] = None
453 self.direct_imports[name] = module
454 self.reverse_alias.pop(name, None)
455 name = name.rpartition(".")[0]

457 def require_name(self, name: str) -> None:
458 while name not in self.direct_imports and "." in name:
459 name = name.rsplit(".", 1)[0]
460 self.required_names.add(name)

471 def import_lines(self) -> list[str]:
472 """The list of required import lines (as strings with python code).
473  
474 In order for a module be included in this output, an indentifier must be both
475 'required' via require_name() and 'imported' via add_import_from()
476 or add_import()
477 """
478 result = []
479  
480 # To summarize multiple names imported from a same module, we collect those
481 # in the `module_map` dictionary, mapping a module path to the list of names that should
482 # be imported from it. the names can also be alias in the form 'original as alias'
483 module_map: Mapping[str, list[str]] = defaultdict(list)
484  
485 for name in sorted(
486 self.required_names,
487 key=lambda n: (self.reverse_alias[n], n) if n in self.reverse_alias else (n, ""),
488 ):
489 # If we haven't seen this name in an import statement, ignore it
490 if name not in self.module_for:
491 continue
492  
493 m = self.module_for[name]
494 if m is not None:
495 # This name was found in a from ... import ...
496 # Collect the name in the module_map
497 if name in self.reverse_alias:
498 name = f"{self.reverse_alias[name]} as {name}"
499 elif name in self.reexports:
500 name = f"{name} as {name}"
501 module_map[m].append(name)
502 else:
503 # This name was found in an import ...
504 # We can already generate the import line
505 if name in self.reverse_alias:
506 source = self.reverse_alias[name]
507 result.append(f"import {source} as {name}\n")
508 elif name in self.reexports:
509 assert "." not in name # Because reexports only has nonqualified names
510 result.append(f"import {name} as {name}\n")
511 else:
512 result.append(f"import {name}\n")
513  
514 # Now generate all the from ... import ... lines collected in module_map
515 for module, names in sorted(module_map.items()):
516 result.append(f"from {module} import {', '.join(sorted(names))}\n")
517 return result

760 def is_private_name(self, name: str, fullname: str | None = None) -> bool:
761 if self._include_private:
762 return False
763 if fullname in self.EXTRA_EXPORTED:
764 return False
765 if name == "_":
766 return False
767 if not name.startswith("_"):
768 return False
769 if self._all_ and name in self._all_:
770 return False
771 if name.startswith("__") and name.endswith("__"):
772 return name in self.IGNORED_DUNDERS
773 return True

775 def should_reexport(self, name: str, full_module: str, name_is_alias: bool) -> bool:
776 if (
777 not name_is_alias
778 and self.module_name
779 and (self.module_name + "." + name) in self.EXTRA_EXPORTED
780 ):
781 # Special case certain names that should be exported, against our general rules.
782 return True
783 if name_is_alias:
784 return False
785 if self.export_less:
786 return False
787 if not self.module_name:
788 return False
789 is_private = self.is_private_name(name, full_module + "." + name)
790 if is_private:
791 return False
792 top_level = full_module.split(".")[0]
793 self_top_level = self.module_name.split(".", 1)[0]
794 if top_level not in (self_top_level, "_" + self_top_level):
795 # Export imports from the same package, since we can't reliably tell whether they
796 # are part of the public API.
797 return False
798 if self._all_:
799 return name in self._all_
800 return True
