30 class TypeState:
31 """This class provides subtype caching to improve performance of subtype checks.
32 It also holds protocol fine grained dependencies.
33  
34 Note: to avoid leaking global state, 'reset_all_subtype_caches()' should be called
35 after a build has finished and after a daemon shutdown. This subtype cache only exists for
36 performance reasons, resetting subtype caches for a class has no semantic effect.
37 The protocol dependencies however are only stored here, and shouldn't be deleted unless
38 not needed any more (e.g. during daemon shutdown).
39 """

41 # '_subtype_caches' keeps track of (subtype, supertype) pairs where supertypes are
42 # instances of the given TypeInfo. The cache also keeps track of whether the check
43 # was done in strict optional mode and of the specific *kind* of subtyping relationship,
44 # which we represent as an arbitrary hashable tuple.
45 # We need the caches, since subtype checks for structural types are very slow.

51 # This contains protocol dependencies generated after running a full build,
52 # or after an update. These dependencies are special because:
53 # * They are a global property of the program; i.e. some dependencies for imported
54 # classes can be generated in the importing modules.
55 # * Because of the above, they are serialized separately, after a full run,
56 # or a full update.
57 # `proto_deps` can be None if after deserialization it turns out that they are
58 # inconsistent with the other cache files (or an error occurred during deserialization).
59 # A blocking error will be generated in this case, since we can't proceed safely.
60 # For the description of kinds of protocol dependencies and corresponding examples,
61 # see _snapshot_protocol_deps.
62 proto_deps: dict[str, set[str]] | None
63  
64 # Protocols (full names) a given class attempted to implement.
65 # Used to calculate fine grained protocol dependencies and optimize protocol
66 # subtype cache invalidation in fine grained mode. For example, if we pass a value
67 # of type a.A to a function expecting something compatible with protocol p.P,
68 # we'd have 'a.A' -> {'p.P', ...} in the map. This map is flushed after every incremental
69 # update.
70 _attempted_protocols: Final[dict[str, set[str]]]
71 # We also snapshot protocol members of the above protocols. For example, if we pass
72 # a value of type a.A to a function expecting something compatible with Iterable, we'd have
73 # 'a.A' -> {'__iter__', ...} in the map. This map is also flushed after every incremental
74 # update. This map is needed to only generate dependencies like <a.A.__iter__> -> <a.A>
75 # instead of a wildcard to avoid unnecessarily invalidating classes.
76 _checked_against_members: Final[dict[str, set[str]]]
77 # TypeInfos that appeared as a left type (subtype) in a subtype check since latest
78 # dependency snapshot update. This is an optimisation for fine grained mode; during a full
79 # run we only take a dependency snapshot at the very end, so this set will contain all
80 # subtype-checked TypeInfos. After a fine grained update however, we can gather only new
81 # dependencies generated from (typically) few TypeInfos that were subtype-checked
82 # (i.e. appeared as r.h.s. in an assignment or an argument in a function call in
83 # a re-checked target) during the update.
84 _rechecked_types: Final[set[TypeInfo]]

228 def _snapshot_protocol_deps(self) -> dict[str, set[str]]:
229 """Collect protocol attribute dependencies found so far from registered subtype checks.
230  
231 There are three kinds of protocol dependencies. For example, after a subtype check:
232  
233 x: Proto = C()
234  
235 the following dependencies will be generated:
236 1. ..., <SuperProto[wildcard]>, <Proto[wildcard]> -> <Proto>
237 2. ..., <B.attr>, <C.attr> -> <C> [for every attr in Proto members]
238 3. <C> -> Proto # this one to invalidate the subtype cache
239  
240 The first kind is generated immediately per-module in deps.py (see also an example there
241 for motivation why it is needed). While two other kinds are generated here after all
242 modules are type checked and we have recorded all the subtype checks. To understand these
243 two kinds, consider a simple example:
244  
245 class A:
246 def __iter__(self) -> Iterator[int]:
247 ...
248  
249 it: Iterable[int] = A()
250  
251 We add <a.A.__iter__> -> <a.A> to invalidate the assignment (module target in this case),
252 whenever the signature of a.A.__iter__ changes. We also add <a.A> -> typing.Iterable,
253 to invalidate the subtype caches of the latter. (Note that the same logic applies to
254 proper subtype checks, and calculating meets and joins, if this involves calling
255 'subtypes.is_protocol_implementation').
256 """
257 deps: dict[str, set[str]] = {}
258 for info in self._rechecked_types:
259 for attr in self._checked_against_members[info.fullname]:
260 # The need for full MRO here is subtle, during an update, base classes of
261 # a concrete class may not be reprocessed, so not all <B.x> -> <C.x> deps
262 # are added.
263 for base_info in info.mro[:-1]:
264 trigger = make_trigger(f"{base_info.fullname}.{attr}")
265 if "typing" in trigger or "builtins" in trigger:
266 # TODO: avoid everything from typeshed
267 continue
268 deps.setdefault(trigger, set()).add(make_trigger(info.fullname))
269 for proto in self._attempted_protocols[info.fullname]:
270 trigger = make_trigger(info.fullname)
271 if "typing" in trigger or "builtins" in trigger:
272 continue
273 # If any class that was checked against a protocol changes,
274 # we need to reset the subtype cache for the protocol.
275 #
276 # Note: strictly speaking, the protocol doesn't need to be
277 # re-checked, we only need to reset the cache, and its uses
278 # elsewhere are still valid (unless invalidated by other deps).
279 deps.setdefault(trigger, set()).add(proto)
280 return deps

282 def update_protocol_deps(self, second_map: dict[str, set[str]] | None = None) -> None:
283 """Update global protocol dependency map.
284  
285 We update the global map incrementally, using a snapshot only from recently
286 type checked types. If second_map is given, update it as well. This is currently used
287 by FineGrainedBuildManager that maintains normal (non-protocol) dependencies.
288 """
289 assert self.proto_deps is not None, "This should not be called after failed cache load"
290 new_deps = self._snapshot_protocol_deps()
291 for trigger, targets in new_deps.items():
292 self.proto_deps.setdefault(trigger, set()).update(targets)
293 if second_map is not None:
294 for trigger, targets in new_deps.items():
295 second_map.setdefault(trigger, set()).update(targets)
296 self._rechecked_types.clear()
297 self._attempted_protocols.clear()
298 self._checked_against_members.clear()
