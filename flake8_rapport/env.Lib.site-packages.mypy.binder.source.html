33 class Frame:
34 """A Frame represents a specific point in the execution of a program.
35 It carries information about the current types of expressions at
36 that point, arising either from assignments to those expressions
37 or the result of isinstance checks. It also records whether it is
38 possible to reach that point at all.
39  
40 This information is not copied into a new Frame when it is pushed
41 onto the stack, so a given Frame only has information about types
42 that were assigned in that frame.
43 """

189 def update_from_options(self, frames: list[Frame]) -> bool:
190 """Update the frame to reflect that each key will be updated
191 as in one of the frames. Return whether any item changes.
192  
193 If a key is declared as AnyType, only update it if all the
194 options are the same.
195 """

197 frames = [f for f in frames if not f.unreachable]
198 changed = False
199 keys = {key for f in frames for key in f.types}
200  
201 for key in keys:
202 current_value = self._get(key)
203 resulting_values = [f.types.get(key, current_value) for f in frames]
204 if any(x is None for x in resulting_values):
205 # We didn't know anything about key before
206 # (current_value must be None), and we still don't
207 # know anything about key in at least one possible frame.
208 continue
209  
210 type = resulting_values[0]
211 assert type is not None
212 declaration_type = get_proper_type(self.declarations.get(key))
213 if isinstance(declaration_type, AnyType):
214 # At this point resulting values can't contain None, see continue above
215 if not all(is_same_type(type, cast(Type, t)) for t in resulting_values[1:]):
216 type = AnyType(TypeOfAny.from_another_any, source_any=declaration_type)
217 else:
218 for other in resulting_values[1:]:
219 assert other is not None
220 type = join_simple(self.declarations[key], type, other)
221 # Try simplifying resulting type for unions involving variadic tuples.
222 # Technically, everything is still valid without this step, but if we do
223 # not do this, this may create long unions after exiting an if check like:
224 # x: tuple[int, ...]
225 # if len(x) < 10:
226 # ...
227 # We want the type of x to be tuple[int, ...] after this block (if it is
228 # still equivalent to such type).

230 type = collapse_variadic_union(type)
231 if isinstance(type, ProperType) and isinstance(type, UnionType):

233 # type when popping a frame.
234 simplified = UnionType.make_union(
235 [t for t in type.items if not isinstance(get_proper_type(t), AnyType)]
236 )
237 if simplified == self.declarations[key]:
238 type = simplified
239 if current_value is None or not is_same_type(type, current_value):

243 self.frames[-1].unreachable = not frames

401 @contextmanager
402 def frame_context(
403 self,
404 *,
405 can_skip: bool,
406 fall_through: int = 1,
407 break_frame: int = 0,
408 continue_frame: int = 0,
409 conditional_frame: bool = False,
410 try_frame: bool = False,
411 ) -> Iterator[Frame]:
412 """Return a context manager that pushes/pops frames on enter/exit.
413  
414 If can_skip is True, control flow is allowed to bypass the
415 newly-created frame.
416  
417 If fall_through > 0, then it will allow control flow that
418 falls off the end of the frame to escape to its ancestor
419 `fall_through` levels higher. Otherwise control flow ends
420 at the end of the frame.
421  
422 If break_frame > 0, then 'break' statements within this frame
423 will jump out to the frame break_frame levels higher than the
424 frame created by this call to frame_context. Similarly for
425 continue_frame and 'continue' statements.
426  
427 If try_frame is true, then execution is allowed to jump at any
428 point within the newly created frame (or its descendants) to
429 its parent (i.e., to the frame that was on top before this
430 call to frame_context).
431  
432 After the context manager exits, self.last_pop_changed indicates
433 whether any types changed in the newly-topmost frame as a result
434 of popping this frame.
435 """
436 assert len(self.frames) > 1
437  
438 if break_frame:
439 self.break_frames.append(len(self.frames) - break_frame)
440 if continue_frame:
441 self.continue_frames.append(len(self.frames) - continue_frame)
442 if try_frame:
443 self.try_frames.add(len(self.frames) - 1)
444  
445 new_frame = self.push_frame(conditional_frame)
446 if try_frame:
447 # An exception may occur immediately
448 self.allow_jump(-1)
449 yield new_frame
450 self.pop_frame(can_skip, fall_through)
451  
452 if break_frame:
453 self.break_frames.pop()
454 if continue_frame:
455 self.continue_frames.pop()
456 if try_frame:
457 self.try_frames.remove(len(self.frames) - 1)
